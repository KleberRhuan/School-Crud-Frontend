2. 📋 useApiQuery - Consultas Inteligentes

2.1. Funcionalidades Principais

O useApiQuery é um wrapper inteligente em torno do TanStack Query que oferece:

  • ✅ Cache automático com configuração otimizada

  • 🔄 Retry policy inteligente com backoff exponencial

  • 📊 Loading states granulares (loading, fetching, refetching)

  • 🚫 Error handling centralizado

  • 🎯 Type safety completa com TypeScript

  • ⚡ Performance otimizada com query keys cacheadas

2.2. Interface API

interface ApiQueryOptions<T = any> {
  endpoint: string
  queryKey: QueryKey
  params?: Record<string, any>
  staleTime?: number
  gcTime?: number
  enabled?: boolean
  retry?: RetryOptions
  retryDelay?: RetryDelayOptions
  onSuccess?: (data: T) => void
  onError?: (error: ApiError) => void
}

function useApiQuery<T>(options: ApiQueryOptions<T>): {
  data: T | undefined
  isLoading: boolean
  isFetching: boolean
  isRefetching: boolean
  error: ApiError | null
  refetch: () => Promise<QueryObserverResult<T>>
  fetchStatus: 'fetching' | 'paused' | 'idle'
}

2.3. Exemplos de Uso

2.3.1. Consulta Básica

const { data, isLoading, error } = useApiQuery({
  endpoint: '/users',
  queryKey: ['users'],
  staleTime: 5 * 60 * 1000, // 5 minutos
})

2.3.2. Consulta com Parâmetros

const { data: user } = useApiQuery({
  endpoint: `/users/${userId}`,
  queryKey: ['user', userId],
  params: { include: 'profile,permissions' },
  enabled: !!userId, // Só executa se userId existir
})

2.3.3. Consulta com Retry Customizado

const { data } = useApiQuery({
  endpoint: '/critical-endpoint',
  queryKey: ['critical'],
  retry: (failureCount, error) => {
    // Não retry em 4xx
    if (error.response?.status >= 400 && error.response?.status < 500) {
      return false
    }
    return failureCount < 3
  },
  retryDelay: (attemptIndex) => Math.min(1000 * 2 ** attemptIndex, 30000)
})

3. ✏️ Mutations - Operações CRUD

3.1. useApiCreate

Hook especializado para criação de recursos:

interface CreateOptions<TData, TVariables> {
  endpoint: string
  invalidateQueries?: QueryKey[]
  onSuccess?: (data: TData, variables: TVariables) => void
  onError?: (error: ApiError, variables: TVariables) => void
  successMessage?: string
}

const createUser = useApiCreate<User, CreateUserData>({
  endpoint: '/users',
  invalidateQueries: [['users']],
  successMessage: 'Usuário criado com sucesso!',
  onSuccess: (newUser) => {
    navigate(`/users/${newUser.id}`)
  }
})

// Uso
createUser.mutate({ name: 'João', email: 'joao@email.com' })
// ou com async/await
const user = await createUser.mutateAsync(userData)

3.2. useApiUpdate

Hook para atualização de recursos existentes:

const updateUser = useApiUpdate<User, UpdateUserData>({
  endpoint: `/users/${userId}`,
  invalidateQueries: [['users'], ['user', userId]],
  successMessage: 'Usuário atualizado!',
  onSuccess: (updatedUser) => {
    // Atualizar cache específico
    queryClient.setQueryData(['user', userId], updatedUser)
  }
})

3.3. useApiDelete

Hook para remoção de recursos:

const deleteUser = useApiDelete({
  endpoint: `/users/${userId}`,
  invalidateQueries: [['users']],
  successMessage: 'Usuário removido!',
  onSuccess: () => {
    // Remover do cache específico
    queryClient.removeQueries({ queryKey: ['user', userId] })
  }
})

3.4. Estados das Mutations

Todos os hooks de mutation retornam estados consistentes:

interface MutationResult<TData, TError, TVariables> {
  mutate: (variables: TVariables) => void
  mutateAsync: (variables: TVariables) => Promise<TData>
  isPending: boolean    // Está executando
  isSuccess: boolean    // Concluída com sucesso
  isError: boolean      // Falhou
  error: TError | null  // Erro da última execução
  data: TData | undefined // Dados da última execução bem-sucedida
  reset: () => void     // Limpa estado
}

4. 🏭 Factory Pattern - createApiMutation

Sistema de factory para criar mutations consistentes:

4.1. Interface da Factory

type HttpMethod = 'GET' | 'POST' | 'PUT' | 'PATCH' | 'DELETE'

interface MutationFactoryOptions<TData = any, TVariables = any> {
  successMessage?: string
  errorMessage?: string
  invalidateQueries?: QueryKey[]
  optimisticUpdate?: {
    queryKey: QueryKey
    updater: (old: any, variables: TVariables) => any
  }
  onSuccess?: (data: TData, variables: TVariables) => void
  onError?: (error: ApiError, variables: TVariables) => void
}

function createApiMutation<TData, TVariables>(
  method: HttpMethod,
  endpoint: string,
  options?: MutationFactoryOptions<TData, TVariables>
): UseMutationResult<TData, ApiError, TVariables>

4.2. Exemplos Práticos

4.2.1. Factory para Entidade User

// 🏭 Definir factory de mutations para usuários
const UserMutations = {
  create: createApiMutation<User, CreateUserData>('POST', '/users', {
    successMessage: 'Usuário criado com sucesso!',
    invalidateQueries: [['users']]
  }),

  update: (id: string) => createApiMutation<User, UpdateUserData>('PUT', `/users/${id}`, {
    successMessage: 'Usuário atualizado!',
    invalidateQueries: [['users'], ['user', id]],
    optimisticUpdate: {
      queryKey: ['user', id],
      updater: (old, variables) => ({ ...old, ...variables })
    }
  }),

  delete: (id: string) => createApiMutation<void, void>('DELETE', `/users/${id}`, {
    successMessage: 'Usuário removido!',
    invalidateQueries: [['users']],
    onSuccess: () => {
      queryClient.removeQueries({ queryKey: ['user', id] })
    }
  }),

  // Ações específicas
  changePassword: (id: string) => createApiMutation<void, ChangePasswordData>(
    'POST',
    `/users/${id}/change-password`,
    {
      successMessage: 'Senha alterada com sucesso!'
    }
  ),

  resetPassword: (id: string) => createApiMutation<void, void>(
    'POST',
    `/users/${id}/reset-password`,
    {
      successMessage: 'Email de recuperação enviado!'
    }
  ),

  toggleStatus: (id: string) => createApiMutation<User, { status: 'active' | 'inactive' }>(
    'PATCH',
    `/users/${id}/status`,
    {
      successMessage: 'Status atualizado!',
      invalidateQueries: [['users'], ['user', id]]
    }
  )
}

// 📝 Uso nos componentes
function UserActions({ user }: { user: User }) {
  const deleteUser = UserMutations.delete(user.id)
  const toggleStatus = UserMutations.toggleStatus(user.id)
  const resetPassword = UserMutations.resetPassword(user.id)

  return (
    <div className="flex gap-2">
      <button
        onClick={() => toggleStatus.mutate({
          status: user.status === 'active' ? 'inactive' : 'active'
        })}
        disabled={toggleStatus.isPending}
      >
        {user.status === 'active' ? 'Desativar' : 'Ativar'}
      </button>

      <button
        onClick={() => resetPassword.mutate()}
        disabled={resetPassword.isPending}
      >
        Resetar Senha
      </button>

      <button
        onClick={() => deleteUser.mutate()}
        disabled={deleteUser.isPending}
        className="btn-danger"
      >
        Deletar
      </button>
    </div>
  )
}

5.2. Retry Policy Customizada

// policies/retryPolicy.ts
export class RetryPolicy {
  static intelligent(failureCount: number, error: any): boolean {
    // Não retry em erros de cliente (4xx)
    if (error?.response?.status >= 400 && error?.response?.status < 500) {
      // Exceto 408 (timeout), 429 (rate limit)
      if (error.response.status === 408 || error.response.status === 429) {
        return failureCount < 3
      }
      return false
    }

    // Retry em erros de servidor (5xx) e rede
    return failureCount < 3
  }

  static aggressive(failureCount: number): boolean {
    return failureCount < 5
  }

  static conservative(failureCount: number): boolean {
    return failureCount < 1
  }

  static exponentialDelay(attemptIndex: number, baseDelay = 1000): number {
    return Math.min(baseDelay * 2 ** attemptIndex, 30000)
  }

  static linearDelay(attemptIndex: number, baseDelay = 1000): number {
    return baseDelay * (attemptIndex + 1)
  }

  static randomJitter(delay: number, jitter = 0.1): number {
    const randomMultiplier = 1 + (Math.random() - 0.5) * 2 * jitter
    return Math.round(delay * randomMultiplier)
  }
}

// Uso em queries
const { data } = useApiQuery({
  endpoint: '/api/data',
  queryKey: ['data'],
  retry: RetryPolicy.intelligent,
  retryDelay: (attemptIndex) =>
    RetryPolicy.randomJitter(
      RetryPolicy.exponentialDelay(attemptIndex, 500)
    )
})

// services/toastService.ts
export interface ToastOptions {
  duration?: number
  position?: 'top-left' | 'top-center' | 'top-right' | 'bottom-left' | 'bottom-center' | 'bottom-right'
  action?: {
    label: string
    onClick: () => void
  }
  persistent?: boolean
}

export interface IToastService {
  success(message: string, options?: ToastOptions): void
  error(message: string, options?: ToastOptions): void
  warning(message: string, options?: ToastOptions): void
  info(message: string, options?: ToastOptions): void
}

6.3. Implementação com Notistack

// services/notistackToastService.ts
import { enqueueSnackbar, SnackbarKey } from 'notistack'
import type { IToastService, ToastOptions } from './toastService'

export class NotistackToastService implements IToastService {
  success(message: string, options?: ToastOptions): void {
    enqueueSnackbar(message, {
      variant: 'success',
      autoHideDuration: options?.duration,
      anchorOrigin: this.getAnchorOrigin(options?.position),
      action: options?.action ? this.createAction(options.action) : undefined,
      persist: options?.persistent
    })
  }

  error(message: string, options?: ToastOptions): void {
    enqueueSnackbar(message, {
      variant: 'error',
      autoHideDuration: options?.duration || 5000,
      anchorOrigin: this.getAnchorOrigin(options?.position),
      action: options?.action ? this.createAction(options.action) : undefined,
      persist: options?.persistent
    })
  }

  warning(message: string, options?: ToastOptions): void {
    enqueueSnackbar(message, {
      variant: 'warning',
      autoHideDuration: options?.duration || 4000,
      anchorOrigin: this.getAnchorOrigin(options?.position)
    })
  }

  info(message: string, options?: ToastOptions): void {
    enqueueSnackbar(message, {
      variant: 'info',
      autoHideDuration: options?.duration || 3000,
      anchorOrigin: this.getAnchorOrigin(options?.position)
    })
  }

  private getAnchorOrigin(position?: ToastOptions['position']) {
    const [vertical, horizontal] = (position || 'top-right').split('-')
    return {
      vertical: vertical as 'top' | 'bottom',
      horizontal: horizontal as 'left' | 'center' | 'right'
    }
  }

  private createAction(action: { label: string; onClick: () => void }) {
    return (snackbarId: SnackbarKey) => (
      <button onClick={() => { action.onClick(); closeSnackbar(snackbarId) }}>
        {action.label}
      </button>
    )
  }
}

6.4. Context Provider

// contexts/ToastContext.tsx
import { createContext, useContext, ReactNode } from 'react'
import { IToastService, NotistackToastService } from '../services/toastService'

const ToastContext = createContext<IToastService | null>(null)

interface ToastProviderProps {
  children: ReactNode
  value?: IToastService
}

export function ToastProvider({ children, value }: ToastProviderProps) {
  const toastService = value || new NotistackToastService()

  return (
    <ToastContext.Provider value={toastService}>
      {children}
    </ToastContext.Provider>
  )
}

export function useToastService(): IToastService {
  const context = useContext(ToastContext)
  if (!context) {
    throw new Error('useToastService deve ser usado dentro de ToastProvider')
  }
  return context
}

7. ⚡ Otimização de Performance

7.1. Query Keys Cacheadas

// config/queryConfig.ts
const queryKeysCache = new Map<string, any[]>()

export function createQueryKeys(entity: string) {
  if (queryKeysCache.has(entity)) {
    return queryKeysCache.get(entity)!
  }

  const keys = {
    all: [entity] as const,
    lists: () => [...keys.all, 'list'] as const,
    list: (filters: Record<string, any>) => [...keys.lists(), filters] as const,
    details: () => [...keys.all, 'detail'] as const,
    detail: (id: string) => [...keys.details(), id] as const,
  }

  queryKeysCache.set(entity, keys)
  return keys
}

// Limpeza do cache (para testes)
export function clearQueryKeysCache() {
  queryKeysCache.clear()
}

// Estatísticas do cache
export function getQueryKeysCacheStats() {
  return {
    size: queryKeysCache.size,
    entities: Array.from(queryKeysCache.keys())
  }
}

// Uso
const userKeys = createQueryKeys('users')
const { data } = useApiQuery({
  endpoint: '/users',
  queryKey: userKeys.list({ status: 'active' })
})

7.2. Prefetching Inteligente

// hooks/usePrefetch.ts
export function usePrefetch() {
  const queryClient = useQueryClient()

  const prefetchUser = useCallback((userId: string) => {
    queryClient.prefetchQuery({
      queryKey: createQueryKeys('users').detail(userId),
      queryFn: () => apiClient.get(`/users/${userId}`),
      staleTime: 5 * 60 * 1000 // 5 minutos
    })
  }, [queryClient])

  const prefetchUsersList = useCallback((filters?: Record<string, any>) => {
    queryClient.prefetchQuery({
      queryKey: createQueryKeys('users').list(filters || {}),
      queryFn: () => apiClient.get('/users', { params: filters }),
      staleTime: 2 * 60 * 1000 // 2 minutos
    })
  }, [queryClient])

  return { prefetchUser, prefetchUsersList }
}

// Uso em componentes
function UserCard({ user }: { user: User }) {
  const { prefetchUser } = usePrefetch()

  return (
    <div
      onMouseEnter={() => prefetchUser(user.id)}
      className="user-card"
    >
      <Link to={`/users/${user.id}`}>
        {user.name}
      </Link>
    </div>
  )
}

import { QueryClient, QueryClientProvider } from '@tanstack/react-query'
import { render, RenderOptions } from '@testing-library/react'

// Query client para testes
function createTestQueryClient() {
  return new QueryClient({
    defaultOptions: {
      queries: {
        retry: false,
        gcTime: 0,
      },
      mutations: {
        retry: false,
      },
    },

// Wrapper para testes
interface TestProviderProps {
  children: ReactNode
  queryClient?: QueryClient
  toastService?: any
}

function TestProvider({ children, queryClient, toastService }: TestProviderProps) {
  const client = queryClient || createTestQueryClient()
  const toast = toastService || createMockToastService()

  return (
    <QueryClientProvider client={client}>
      <ToastProvider value={toast}>
        {children}
      </ToastProvider>
    </QueryClientProvider>
  )
}

// Render customizado
interface CustomRenderOptions extends Omit<RenderOptions, 'wrapper'> {
  queryClient?: QueryClient
  toastService?: any
}

export function renderWithProviders(
  ui: ReactElement,
  options: CustomRenderOptions = {}
) {
  const { queryClient, toastService, ...renderOptions } = options

  function Wrapper({ children }: { children: ReactNode }) {
    return (
      <TestProvider queryClient={queryClient} toastService={toastService}>
        {children}
      </TestProvider>
    )
  }

  return render(ui, { wrapper: Wrapper, ...renderOptions })
}

9.2. Performance Monitoring

// utils/performanceMonitor.ts
export class QueryPerformanceMonitor {
  private static metrics = new Map<string, number[]>()

  static recordQueryTime(queryKey: string, duration: number) {
    if (!this.metrics.has(queryKey)) {
      this.metrics.set(queryKey, [])
    }
    this.metrics.get(queryKey)!.push(duration)
  }

  static getAverageTime(queryKey: string): number {
    const times = this.metrics.get(queryKey) || []
    return times.reduce((sum, time) => sum + time, 0) / times.length
  }

  static getSlowQueries(threshold = 1000): Array<{ key: string; avgTime: number }> {
    return Array.from(this.metrics.entries())
      .map(([key, times]) => ({
        key,
        avgTime: this.getAverageTime(key)
      }))
      .filter(({ avgTime }) => avgTime > threshold)
      .sort((a, b) => b.avgTime - a.avgTime)
  }

  static generateReport(): string {
    const slowQueries = this.getSlowQueries()
    return `
📊 Query Performance Report
==========================
Total Queries: ${this.metrics.size}
Slow Queries (>1s): ${slowQueries.length}

${slowQueries.map(({ key, avgTime }) =>
  `🐌 ${key}: ${avgTime.toFixed(2)}ms avg`
).join('\n')}
    `
  }
}

10.1. Migração de Hooks Legados

// ❌ Antes (padrão antigo)
import { useState, useEffect } from 'react'
import axios from 'axios'

function useOldApiHook(url: string) {
  const [data, setData] = useState(null)
  const [loading, setLoading] = useState(false)
  const [error, setError] = useState(null)

  useEffect(() => {
    setLoading(true)
    axios.get(url)
      .then(response => setData(response.data))
      .catch(err => setError(err))
      .finally(() => setLoading(false))
  }, [url])

  return { data, loading, error }
}

// ✅ Depois (hooks refatorados)
import { useApiQuery } from '@/hooks/api'

function useNewApiHook(endpoint: string, queryKey: QueryKey) {
  return useApiQuery({
    endpoint,
    queryKey,
    staleTime: 5 * 60 * 1000
  })
}
