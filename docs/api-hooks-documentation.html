  endpoint: string

  params?: Record<string, any>

// policies/retryPolicy.ts
export class RetryPolicy {
  static intelligent(failureCount: number, error: any): boolean {
    // N√£o retry em erros de cliente (4xx)
    if (error?.response?.status >= 400 && error?.response?.status < 500) {
      // Exceto 408 (timeout), 429 (rate limit)
      if (error.response.status === 408 || error.response.status === 429) {
        return failureCount < 3
      }
      return false
    }

    // Retry em erros de servidor (5xx) e rede
    return failureCount < 3
  }

  static aggressive(failureCount: number): boolean {
    return failureCount < 5
  }

  static conservative(failureCount: number): boolean {
    return failureCount < 1
  }

  static exponentialDelay(attemptIndex: number, baseDelay = 1000): number {
    return Math.min(baseDelay * 2 ** attemptIndex, 30000)
  }

  static linearDelay(attemptIndex: number, baseDelay = 1000): number {
    return baseDelay * (attemptIndex + 1)
  }

  static randomJitter(delay: number, jitter = 0.1): number {
    const randomMultiplier = 1 + (Math.random() - 0.5) * 2 * jitter
    return Math.round(delay * randomMultiplier)
  }
}

// Uso em queries
const { data } = useApiQuery({
  endpoint: '/api/data',
  queryKey: ['data'],
  retry: RetryPolicy.intelligent,
  retryDelay: (attemptIndex) =>
    RetryPolicy.randomJitter(
      RetryPolicy.exponentialDelay(attemptIndex, 500)
    )
})

6.3. Implementa√ß√£o com Notistack

// services/notistackToastService.ts
import { enqueueSnackbar, SnackbarKey } from 'notistack'
import type { IToastService, ToastOptions } from './toastService'

export class NotistackToastService implements IToastService {
  success(message: string, options?: ToastOptions): void {
    enqueueSnackbar(message, {
      variant: 'success',
      autoHideDuration: options?.duration,
      anchorOrigin: this.getAnchorOrigin(options?.position),
      action: options?.action ? this.createAction(options.action) : undefined,
      persist: options?.persistent
    })
  }

  error(message: string, options?: ToastOptions): void {
    enqueueSnackbar(message, {
      variant: 'error',
      autoHideDuration: options?.duration || 5000,
      anchorOrigin: this.getAnchorOrigin(options?.position),
      action: options?.action ? this.createAction(options.action) : undefined,
      persist: options?.persistent
    })
  }

  warning(message: string, options?: ToastOptions): void {
    enqueueSnackbar(message, {
      variant: 'warning',
      autoHideDuration: options?.duration || 4000,
      anchorOrigin: this.getAnchorOrigin(options?.position)
    })
  }

  info(message: string, options?: ToastOptions): void {
    enqueueSnackbar(message, {
      variant: 'info',
      autoHideDuration: options?.duration || 3000,
      anchorOrigin: this.getAnchorOrigin(options?.position)
    })
  }

  private getAnchorOrigin(position?: ToastOptions['position']) {
    const [vertical, horizontal] = (position || 'top-right').split('-')
    return {
      vertical: vertical as 'top' | 'bottom',
      horizontal: horizontal as 'left' | 'center' | 'right'
    }
  }

  private createAction(action: { label: string; onClick: () => void }) {
    return (snackbarId: SnackbarKey) => (
      <button onClick={() => { action.onClick(); closeSnackbar(snackbarId) }}>
        {action.label}
      </button>
    )
  }
}

}

// config/queryConfig.ts
const queryKeysCache = new Map<string, any[]>()

export function createQueryKeys(entity: string) {
  if (queryKeysCache.has(entity)) {
    return queryKeysCache.get(entity)!
  }

  const keys = {
    all: [entity] as const,
    lists: () => [...keys.all, 'list'] as const,
    list: (filters: Record<string, any>) => [...keys.lists(), filters] as const,
    details: () => [...keys.all, 'detail'] as const,
    detail: (id: string) => [...keys.details(), id] as const,
  }

  queryKeysCache.set(entity, keys)
  return keys
}

// Limpeza do cache (para testes)
export function clearQueryKeysCache() {
  queryKeysCache.clear()
}

// Estat√≠sticas do cache
export function getQueryKeysCacheStats() {
  return {
    size: queryKeysCache.size,
    entities: Array.from(queryKeysCache.keys())
  }
}

// hooks/usePrefetch.ts
export function usePrefetch() {
  const queryClient = useQueryClient()

  const prefetchUser = useCallback((userId: string) => {
    queryClient.prefetchQuery({
      queryKey: createQueryKeys('users').detail(userId),
      queryFn: () => apiClient.get(`/users/${userId}`),
      staleTime: 5 * 60 * 1000 // 5 minutos
    })
  }, [queryClient])

  const prefetchUsersList = useCallback((filters?: Record<string, any>) => {
    queryClient.prefetchQuery({
      queryKey: createQueryKeys('users').list(filters || {}),
      queryFn: () => apiClient.get('/users', { params: filters }),
      staleTime: 2 * 60 * 1000 // 2 minutos
    })
  }, [queryClient])

  return { prefetchUser, prefetchUsersList }
}

// utils/performanceMonitor.ts
export class QueryPerformanceMonitor {
  private static metrics = new Map<string, number[]>()

  static recordQueryTime(queryKey: string, duration: number) {
    if (!this.metrics.has(queryKey)) {
      this.metrics.set(queryKey, [])
    }
    this.metrics.get(queryKey)!.push(duration)
  }

  static getAverageTime(queryKey: string): number {
    const times = this.metrics.get(queryKey) || []
    return times.reduce((sum, time) => sum + time, 0) / times.length
  }

  static getSlowQueries(threshold = 1000): Array<{ key: string; avgTime: number }> {
    return Array.from(this.metrics.entries())
      .map(([key, times]) => ({
        key,
        avgTime: this.getAverageTime(key)
      }))
      .filter(({ avgTime }) => avgTime > threshold)
      .sort((a, b) => b.avgTime - a.avgTime)
  }

  static generateReport(): string {
    const slowQueries = this.getSlowQueries()
    return `

${slowQueries.map(({ key, avgTime }) =>
  `üêå ${key}: ${avgTime.toFixed(2)}ms avg`
).join('\n')}
    `
  }
}
